stts_press_tour = {
	icon = "gfx/interface/icons/generic_icons/ig_bolster.dds"
	clicksound = "event:/SFX/UI/Global/promote"

	potential = {
		age >= define:NCharacters|ADULT_AGE
		owner = {
			exists = this
			has_technology_researched = mass_communication
		}
	}

	possible = {
		trigger_if = {
			limit = {owner ?= {has_law = law_type:law_protected_speech}}
			owner ?= {
				authority >= 100
			}
		}
		trigger_else_if = {
			limit = {owner ?= {has_law = law_type:law_right_of_assembly }}
			owner ?= {
				authority >= 75
			}
		}
		trigger_else_if = {
			limit = {owner ?= {has_law = law_type:law_censorship }}
			owner ?= {
				authority >= 50
			}
		}
		trigger_else = {
			owner ?= {
				authority >= 25
			}
		}
		custom_tooltip = {
			text = stts_no_current_press_tour
			NOR = {
				has_modifier = stts_character_smear_campaign_modifier
				has_modifier = stts_character_press_tour_modifier
			}
		}
	}

	effect = {
		add_modifier = {
			name = stts_character_press_tour_modifier
			years = 10
			is_decaying = yes
		}
		interest_group = {
			add_modifier = {
				name = stts_ig_press_tour_modifier
				years = 5
			}
		}
		if = {
			limit = {owner ?= {has_law = law_type:law_protected_speech}}
			owner = {
				add_modifier = {
					name = stts_press_authority_modifier
					years = 5
					multiplier = 4
				}
			}
		}
		else_if = {
			limit = {owner ?= {has_law = law_type:law_right_of_assembly}}
			owner = {
				add_modifier = {
					name = stts_press_authority_modifier
					years = 5
					multiplier = 3
				}
			}
		}
		else_if = {
			limit = {owner ?= {has_law = law_type:law_censorship}}
			owner = {
				add_modifier = {
					name = stts_press_authority_modifier
					years = 5
					multiplier = 2
				}
			}
		}
		else = {
			owner = {
				add_modifier = {
					name = stts_press_authority_modifier
					years = 5
					multiplier = 1
				}
			}
		}

	}

	ai_chance = {
		value = 0
	}
}
