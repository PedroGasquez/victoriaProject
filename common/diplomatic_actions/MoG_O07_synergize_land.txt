laws_synergize_land = {
	groups= {
		overlord
	}
	
	requires_approval = no
	should_notify_third_parties = no
	
	potential = {
		NOT = { has_game_rule = sis_synergize_disabled }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
			is_direct_subject_of = root
			NOT = { scope:target_country.active_law:lawgroup_land_reform.type = root.active_law:lawgroup_land_reform.type }
		}
		is_player = yes
	}
	
	possible = {
		is_revolutionary = no
		in_default = no
		NOT = { has_war_with = scope:target_country }
		NOT = {
			any_country = { has_diplomatic_pact = { who = root type = laws_synergize_land  } } # Only 1 at a time.
		}
		has_diplomatic_relevance = scope:target_country
		active_synergize_land = no
#		bureaucracy > 0
	}
	
	accept_effect = {
		scope:target_country = {
			set_variable = {
				name = synergize_land_progress
				value = 0
			}
		}
		add_modifier = {
			name = overlord_syn_cost_land
			multiplier = {
				add = 1
				scope:target_country = { multiply = overlord_synergize_cost_standard }
				if = {
					limit = { NOT = { scope:target_country.active_law:lawgroup_bureaucracy.type = root.active_law:lawgroup_bureaucracy.type } }
					multiply = 2
				}
			}
		}
		set_variable = { #used to count the number of agreements
			name = synergize_land_active
		}
	}
	
	pact = {
		cost = 100
	
		is_two_sided_pact = no
		show_in_outliner = yes
		
		actor_can_break = {
			always = yes
		}
		
		weekly_effect = {
			custom_tooltip = {
				text = laws_synergize_land_weekly_effect
				random_list = { 
					4 = {}
					1 = {
						# Updates the modifier cost.
						remove_modifier = overlord_syn_cost_land
						add_modifier = {
							name = overlord_syn_cost_land
							multiplier = {
								add = 1
								scope:target_country = { add = overlord_synergize_cost_standard }
								if = {
									limit = { NOT = { scope:target_country.active_law:lawgroup_bureaucracy.type = root.active_law:lawgroup_bureaucracy.type } }
									multiply = 2
								}
							}
						}
						scope:target_country = {
							change_variable = { name = synergize_land_progress add = 1 }
						}
						if = {
							limit = {
								scope:target_country = {
									has_variable = synergize_land_progress
									var:synergize_land_progress > 7.5 # Just in case of rounding edge-cases.
								}
							}
							mog_sis_synergy_complete_effect = {
								COST_MODIFIER = overlord_syn_cost_land
								ACTIVE_VARIABLE = synergize_land_active
								SYNERGY_VARIABLE = synergize_land_progress
								DIPLO_PACT = laws_synergize_land
								ACTIVE_LAW = active_law:lawgroup_land_reform.type
							}
						}
					}
				}
			}
		}
		
		manual_break_effect = {
			root = { 
				remove_modifier = overlord_syn_cost_land
				remove_variable = synergize_land_active
			}
		}
		
		auto_break_effect = {
			root = { 
				remove_modifier = overlord_syn_cost_land
				remove_variable = synergize_land_active
			}
		}
		
		requirement_to_maintain = {
			trigger = {
				has_diplomatic_relevance = scope:target_country
			}
		}
		
		requirement_to_maintain = {
			trigger = {
				NOT = { scope:target_country.active_law:lawgroup_land_reform.type = root.active_law:lawgroup_land_reform.type }
			}
		}
		
		requirement_to_maintain = {
			trigger = {
				in_default = no
			}
		}
		
		requirement_to_maintain = {
			trigger = {
				custom_tooltip = {
					text = DIPLO_ACTION_RELATIONS_CORDIAL_OR_GREATER
					scope:target_country.relations:root >= relations_threshold:cordial
				}		
			}
			show_about_to_break_warning = {
				scope:target_country.relations:root <= 23 # this is: cordial + 3
				scope:target_country.relations_change_rate:root > 0
			}
		}
		
		requirement_to_maintain = {
			trigger = {
				scope:target_country = { liberty_desire <= 80 }
			}
			show_about_to_break_warning = {
				scope:target_country = { liberty_desire >= 75 }
			}
		}
		
		requirement_to_maintain = {
			trigger = {
				custom_tooltip = {
					text = DIPLOMATIC_PACT_REQUIREMENT_AT_PEACE
					NOT = { has_war_with = scope:target_country }
				}			
			}
		}
		
		requirement_to_maintain = {
			trigger = {
				custom_tooltip = {
					text = "DIPLOMATIC_PACT_REQUIREMENT_NOT_DECENTRALIZED"
					NOT = { is_country_type = decentralized }
					scope:target_country = {
						NOT = { is_country_type = decentralized }
					}				
				}
			}
		}
	}
}
