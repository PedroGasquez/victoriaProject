## Did you know that non-Germans can become the Confederation of the Rhine through journal entries? Did you know that they won't get North and South German as Primary Cultures? ##

on_monthly_pulse_country = {
	on_actions = {
		EC_rhine_confederation_culture_check
	}
}

EC_rhine_confederation_culture_check = {

	effect = {
		every_country = {
			limit = {
				exists = c:RHN
				c:RHN = THIS
				NOT = {
					c:RHN = {
						country_has_primary_culture = cu:north_german
						country_has_primary_culture = cu:south_german
					}
				}
			}
			add_primary_culture = cu:north_german
			add_primary_culture = cu:south_german
		}
	}
	
}
