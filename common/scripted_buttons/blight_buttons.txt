	name = "irish_famine_subsidize_grain_shipments"
	desc = "irish_famine_subsidize_grain_shipments_desc"

	ai_chance = {
		value = 100
	}

	visible = {
		NOT = {
			has_variable = sent_grain_to_ireland
		}
	}

	possible = {
		OR = {
			ig:ig_landowners = {
				is_in_government = no
			}
			NOT = {
				has_law = law_type:law_mercantilism
			}
		}
	}

	effect = {
		set_variable = sent_grain_to_ireland
		irish_famine_ease = yes
		ig:ig_landowners = {
			add_modifier = {
				name = ig_helped_irish_negative_modifier
				years = 4
			}
		}
		add_modifier = {
			name = irish_grain_subsidized
			years = 2
		}
	}
}
