ite_japanese_ig_traits_effect = {
	ig:ig_rural_folk = {
		set_ig_trait = ig_trait:ite_ig_trait_ikki
	}
	ig:ig_armed_forces = {
		if = {
			limit = { has_ideology = ideology:ideology_bakufu }
			set_ig_trait = ig_trait:ite_ig_trait_ronin
		} else = {
			set_ig_trait = ig_trait:ite_ig_trait_interservice_rivalry
		}
	}
	ig:ig_intelligentsia = {
		set_ig_trait = ig_trait:ite_ig_trait_rangaku
	}
}

ite_central_asian_rural_folk_effect = {
	ig:ig_rural_folk = {
		set_ig_trait = ig_trait:ite_ig_trait_tribal_militias
		set_ig_trait = ig_trait:ite_ig_trait_alienated_tribes
	}
	#special Pashtun trait
	if = {
		limit = { 
			country_has_primary_culture = cu:pashtun
		}
		ig:ig_rural_folk = {
			set_ig_trait = ig_trait:ite_ig_trait_pashtunwali
		}
	}
}
