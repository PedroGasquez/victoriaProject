blight_zone_catastrophic = {
	OR = {
		state_is_in_ireland = yes
		state_region = s:STATE_HIGHLANDS
	}
}

blight_zone_severe = {
	OR = {
		region = sr:region_rhine
		region = sr:region_north_germany
		state_region = s:STATE_JUTLAND
		state_region = s:STATE_ZEALAND	
	}
}

blight_zone_regular = {
	OR = {
		region = sr:region_france
		region = sr:region_england
		state_region = s:STATE_LOWLANDS
		state_region = s:STATE_SCANIA
		state_region = s:STATE_GOTALAND
		state_region = s:STATE_SVEALAND
	}
}

blight_zone_usa = {
	OR = {
		state_region = s:STATE_ONTARIO
		state_region = s:STATE_MICHIGAN
		region = sr:region_new_england
	}
	NOT = {
		state_region = s:STATE_BERMUDA
	}
}

blight_zone_europe = {
	OR = {
		blight_zone_catastrophic = yes
		blight_zone_severe = yes
		blight_zone_regular = yes
	}
}

state_is_in_ireland = {
	OR = {
		state_region = s:STATE_LEINSTER
		state_region = s:STATE_MUNSTER
		state_region = s:STATE_ULSTER
		state_region = s:STATE_CONNAUGHT
	}
}

grain_failure_zone = {
	OR = {
		state_is_in_europe = yes
	}
}

blight_zone_regions = {
	OR = {
		blight_zone_catastrophic_regions = yes
		blight_zone_usa_region = yes
		blight_zone_europe_regions = yes
	}
}

blight_zone_catastrophic_regions = {
	OR = {
		any_scope_state = {
			OR = {
				state_is_in_ireland = yes
				state_region = s:STATE_HIGHLANDS
			}
		}
	}
}

blight_zone_usa_region = {
	any_scope_state = {
		OR = {
			state_region = s:STATE_ONTARIO
			state_region = s:STATE_MICHIGAN
			region = sr:region_new_england
		}
	}
	NOT = {
		any_scope_state = {
			state_region = s:STATE_BERMUDA
		}
	}
}

blight_zone_europe_regions = {
	OR = {
		region = sr:region_france
		region = sr:region_england
		region = sr:region_rhine
		region = sr:region_north_germany
		any_scope_state = {
			OR = {
				state_region = s:STATE_LOWLANDS
				state_region = s:STATE_SCANIA
				state_region = s:STATE_GOTALAND
				state_region = s:STATE_SVEALAND
				state_region = s:STATE_JUTLAND
				state_region = s:STATE_ZEALAND	
			}
		}
	}
}
